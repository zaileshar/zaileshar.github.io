RTL Example (snippet)

/* Simple illustrative Verilog MAC snippet (concept) */ module mac_simple #(parameter W=32)( input logic clk, rst, input logic [W-1:0] a, b, input logic acc_en, output logic [W-1:0] y ); logic [2*W-1:0] product; logic [2*W-1:0] acc; always_ff @(posedge clk or posedge rst) begin if (rst) begin acc <= '0; product <= '0; y <= '0; end else begin product <= a * b; // multiplier (instantiated in real design) if (acc_en) acc <= acc + product[W-1:0]; y <= acc[W-1:0]; end end endmodule
